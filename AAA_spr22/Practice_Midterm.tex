\begin{homeworkProblem}
    Let $H \leq G$ and $A \leq G$ such that 
    $\lvert G:H \rvert = 3$ and $\lvert A \rvert = 85$.
    Prove that 
    \begin{enumerate}
        \item $A \leq H$.
        \item $\cap_{g \in G} g^{-1} H g \neq 1$.
    \end{enumerate}
    

\end{homeworkProblem}
